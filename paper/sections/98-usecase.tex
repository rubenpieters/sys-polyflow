\section{Scenarios}

\subsection{Main Scenario}

In typescript:

\begin{lstlisting}

interface F {
  "true": "true",
  "false": "false",
}

function f<T extends "true" | "false">(x: T): F[T] {
  if (x === "true") {
    return "true";
  } else {
    return "false";
  }
}
\end{lstlisting}
